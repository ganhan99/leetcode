```python
class Solution1:
    def reverse(self, x: int) -> int:
        lis=str(x)
        lis=list(lis)
        if lis[0]=='-':
            lis1=lis[1:]
            lis1.reverse()
            lis1=list('-')+lis1
        else:
            lis1=lis[:]
            lis1.reverse()
        result =int( "".join(lis1))
        if -2**31<result<2**31-1:
            return result
        return 0
```
